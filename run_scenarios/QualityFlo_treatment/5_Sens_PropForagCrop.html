compute_prop_forag_crop_sens <- function(quality_flo,
                                         alphas = alphas) {

  
  # ratio crop / (crop + nat) -- alpha independant
  denom <- quality_flo$quality_crop + quality_flo$quality_nat
  ratio_crop <- ifelse(
    denom == 0 | is.na(denom),
    0,
    quality_flo$quality_crop / denom
  )
  
  res_list <- lapply(alphas, function(a) {
    prop_forag_crop <- a * ratio_crop
    
    data.frame(
      date            = quality_flo$date,
      quality_crop    = quality_flo$quality_crop,
      quality_nat     = quality_flo$quality_nat,
      alpha           = a,
      ratio_crop      = ratio_crop,
      prop_forag_crop = prop_forag_crop
    )
  })
  
  do.call(rbind, res_list)
}
